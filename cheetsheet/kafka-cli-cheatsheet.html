📋
Topics
Tạo · Liệt kê · Xem · Xoá
Liệt kê tất cả topics
$ docker exec kafka \
  kafka-topics \
  --bootstrap-server kafka:29092 \
  --list
Tạo topic mới
$ docker exec kafka \
  kafka-topics \
  --bootstrap-server kafka:29092 \
  --create --topic my-topic \
  --partitions 3 \
  --replication-factor 1
Xem chi tiết một topic
$ docker exec kafka \
  kafka-topics \
  --bootstrap-server kafka:29092 \
  --describe \
  --topic cdc_lab.shop.customers
Hiển thị số partition, replication factor, leader
Xoá topic
$ docker exec kafka \
  kafka-topics \
  --bootstrap-server kafka:29092 \
  --delete --topic my-topic
📤
Producer
Gửi message vào topic
Gửi message thủ công (interactive)
$ docker exec -it kafka \
  kafka-console-producer \
  --bootstrap-server kafka:29092 \
  --topic my-topic
Gõ message → Enter gửi. Ctrl+C để thoát.
Gửi message có key
$ docker exec -it kafka \
  kafka-console-producer \
  --bootstrap-server kafka:29092 \
  --topic my-topic \
  --property parse.key=true \
  --property key.separator=:
Nhập dạng key:value mỗi dòng
Pipe file vào topic
$ cat data.json | \
  docker exec -i kafka \
  kafka-console-producer \
  --bootstrap-server kafka:29092 \
  --topic my-topic
📥
Consumer
Đọc message từ topic
Đọc từ đầu (toàn bộ history)
$ docker exec kafka \
  kafka-console-consumer \
  --bootstrap-server kafka:29092 \
  --topic cdc_lab.shop.customers \
  --from-beginning
Đọc toàn bộ message đã có. Ctrl+C để dừng.
Đọc kèm key và timestamp
$ docker exec kafka \
  kafka-console-consumer \
  --bootstrap-server kafka:29092 \
  --topic cdc_lab.shop.orders \
  --from-beginning \
  --property print.key=true \
  --property print.timestamp=true
Output: Timestamp\tKey\tValue
Giới hạn số message đọc
$ docker exec kafka \
  kafka-console-consumer \
  --bootstrap-server kafka:29092 \
  --topic my-topic \
  --from-beginning \
  --max-messages 10
Đọc đúng 10 message rồi tự thoát
Đọc với consumer group
$ docker exec kafka \
  kafka-console-consumer \
  --bootstrap-server kafka:29092 \
  --topic my-topic \
  --group lab-consumer-group
Offset được lưu → chạy lại tiếp tục từ chỗ cũ
👥
Consumer Groups
Quản lý group · Xem lag
Liệt kê tất cả consumer groups
$ docker exec kafka \
  kafka-consumer-groups \
  --bootstrap-server kafka:29092 \
  --list
Xem lag của một group
$ docker exec kafka \
  kafka-consumer-groups \
  --bootstrap-server kafka:29092 \
  --describe \
  --group lab-consumer-group
Cột LAG: số message chưa xử lý. LAG=0 là đang bắt kịp.
Reset offset về đầu
$ docker exec kafka \
  kafka-consumer-groups \
  --bootstrap-server kafka:29092 \
  --group lab-consumer-group \
  --topic my-topic \
  --reset-offsets \
  --to-earliest \
  --execute
Lưu ý: group phải đang không có consumer nào chạy
🔌
Kafka Connect REST API
Quản lý Debezium connectors
Liệt kê connectors đang chạy
$ curl -s \
  http://localhost:8083/connectors
Xem trạng thái connector
$ curl -s \
  http://localhost:8083/connectors/\
  postgres-cdc-connector/status
Đăng ký connector từ file JSON
$ curl -X POST \
  -H "Content-Type: application/json" \
  -d @connectors/postgres-cdc-connector.json \
  http://localhost:8083/connectors
Restart connector bị lỗi
$ curl -X POST \
  http://localhost:8083/connectors/\
  postgres-cdc-connector/restart
Xoá connector
$ curl -X DELETE \
  http://localhost:8083/connectors/\
  postgres-cdc-connector
Liệt kê connector plugins đã cài
$ curl -s \
  http://localhost:8083/connector-plugins \
  | python3 -m json.tool

Cluster & Config
Thông tin broker · Cấu hình
Xem thông tin broker
$ docker exec kafka \
  kafka-broker-api-versions \
  --bootstrap-server kafka:29092
Xem metadata cluster
$ docker exec kafka \
  kafka-metadata-quorum \
  --bootstrap-server kafka:29092 \
  --command-config /dev/null \
  describe --status
Xem config của topic
$ docker exec kafka \
  kafka-configs \
  --bootstrap-server kafka:29092 \
  --entity-type topics \
  --entity-name my-topic \
  --describe
Xem retention, cleanup policy, v.v.
Xem offset hiện tại của topic
$ docker exec kafka \
  kafka-get-offsets \
  --bootstrap-server kafka:29092 \
  --topic cdc_lab.shop.products
Kết quả: topic:partition:offset

PostgreSQL — Thao tác CDC
Kết nối, kiểm tra replication, thực hành INSERT / UPDATE / DELETE để trigger CDC events
Kết nối vào PostgreSQL
$ docker exec -it postgres \
  psql -U labuser -d labdb
Kiểm tra WAL level (phải là logical)
SHOW wal_level;
-- kết quả: logical ✓
Xem replication slots
SELECT slot_name, active
FROM pg_replication_slots;
INSERT → CDC op: "c" (create)
INSERT INTO shop.customers
  (full_name, email, phone)
VALUES
  ('Nguyen Van X',
   'x@example.com',
   '0900000001');
UPDATE → CDC op: "u" (update)
UPDATE shop.products
SET price = 25000000,
    stock = 8
WHERE id = 1;
DELETE → CDC op: "d" (delete)
DELETE FROM shop.customers
WHERE email = 'x@example.com';
Xem publications (Debezium tạo tự động)
SELECT pubname, puballtables
FROM pg_publication;
Debezium tạo publication debezium khi kết nối lần đầu
Xoá replication slot (nếu bị kẹt)
SELECT pg_drop_replication_slot(
  'debezium_lab_slot'
);
Dùng khi connector bị lỗi và cần đăng ký lại từ đầu
Xem tổng số bản ghi mỗi bảng
SELECT 'customers',
  COUNT(*) FROM shop.customers
UNION ALL SELECT 'products',
  COUNT(*) FROM shop.products;

Docker Compose — Lệnh quản lý lab
Chạy từ thư mục gốc của repo
Khởi động toàn bộ stack
$ docker compose up -d
Xem trạng thái containers
$ docker compose ps
Xem log realtime
$ docker compose logs -f
# Hoặc 1 service cụ thể:
$ docker compose logs -f kafka
Restart một service
$ docker compose restart \
  kafka-connect
Dừng — giữ nguyên data
$ docker compose down
Reset hoàn toàn (xoá data)
$ docker compose down -v
Cảnh báo: mất toàn bộ data Kafka + PostgreSQL
Kiểm tra sức khoẻ hệ thống
$ bash scripts/check-status.sh
Đăng ký / cập nhật connector
$ bash scripts/register-connector.sh
